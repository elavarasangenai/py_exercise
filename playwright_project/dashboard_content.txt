    function validateInputalphanumeric(event) {
        // Get the character code of the pressed key
        var charCode = event.charCode || event.keyCode;
        // Allowed characters: alphanumeric, and the specified symbols
        var allowedChars = "!@#%^&*()-_+=\\|,.?/:abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";

        // Check if the pressed key is an allowed character or a space
        if (allowedChars.indexOf(String.fromCharCode(charCode)) === -1 && charCode !== 32) {
            // Prevent the default action (typing the disallowed character)
            event.preventDefault();
            // Display an error message
            document.getElementById("errorMessage").textContent = "Only allowed characters are: ! @ # % ^ & * ( ) - _ + = \\ | , . ? / : and alphanumeric.";
        } else {
            // Clear any previous error message
            document.getElementById("errorMessage").textContent = "";
        }
    }

    /**
     * Handles the change event for the branch dropdown and updates the department dropdown based on the selected branch.
     *
     * When a branch is selected from the `.departmentByBranchselect` dropdown, this function triggers an AJAX request
     * to fetch the list of departments associated with that branch. The list is then used to populate the department
     * dropdown. If no branch is selected, the department dropdown is reset to show "All" as the default option.
     *
     * @event  change
     * @param  {string} branch_id The ID of the selected branch
     */
    $('.departmentByBranchselect').on('change', function() {
        var branch_id = $(this).val();  // Get the selected branch ID
        if (branch_id !== '') {
            $('.departmentByBranchoption').html('<option value=" ">Processing...</option>');
            $.ajax({
                type: "GET",  // HTTP method
                url: 'https://web.nithrapeople.com/staff/get-department-list-by-branch',  // URL for AJAX request
                data: {'branch_id': branch_id},  // Send the selected branch ID
                cache: false,  // Disable cache to ensure fresh data is fetched
                dataType: 'json',  // Expect JSON response
                success: function(res) {
                    // Populate the department dropdown with the received department data
                    $('.departmentByBranchoption').html(res.department);
                },
                error: function(xhr, status, error) {
                    // Handle errors (optional)
                    console.error('Error fetching departments:', error);
                }
            });
        } else {
            // If no branch is selected, reset the department dropdown to show "All"
            $('.departmentByBranchoption').html('<option value=" ">All</option>');
        }
    });

     * Common function to apply jQuery form validation.
     *
     * @param  {string} formSelector - The CSS selector for the form to be validated.
     * @param  {object} rules - The validation rules for form fields.
     * @param  {object} messages - Custom error messages for the validation rules.
     */
    function commonValidateForm(formSelector, rules, messages) {
        // Apply validation to the form using the jQuery validate method
        $(formSelector).validate({

            // Define the validation rules for each form field
            rules: rules,

            // Define the custom error messages for each validation rule
            messages: messages,

            /**
             * Defines where to place error messages in the form.
             * By default, it will place error messages after the form element.
             *
             * @param  {object} error - The error message element.
             * @param  {object} element - The form element that failed validation.
             */
            errorPlacement: function(error, element) {
                error.insertAfter(element); // Insert the error message after the input field
            },

            /**
             * Defines how to style the form elements when they are invalid.
             * Adds the 'is-invalid' class to invalid elements for styling.
             *
             * @param  {object} element - The form element that is invalid.
             */
            highlight: function(element) {
                $(element).addClass('is-invalid'); // Add 'is-invalid' class to highlight invalid fields
            },

             * Defines how to remove styling when the form element becomes valid.
             * Removes the 'is-invalid' class when validation passes.
             *
             * @param  {object} element - The form element that is valid.
             */
            unhighlight: function(element) {
                $(element).removeClass('is-invalid'); // Remove 'is-invalid' class when the field is valid
            },

             * Defines what to do when the form is successfully validated.
             * In this case, it submits the form, but custom actions can be performed.
             *
             * @param  {object} form - The form element that passed validation.
             */

                form.submit(); // Submit the form or perform any other action on successful validation
            }
        });

        function datePicker() {
            return {
                currentDate: new Date(),
                formattedDate: '',

                init() {
                    this.updateFormattedDate();
                },

                updateFormattedDate() {
                    const options = {
                        day: '2-digit',
                        month: '2-digit',
                        year: 'numeric'
                    };
                    this.formattedDate = this.currentDate.toLocaleDateString('en-GB', options);
                },

                updateDate() {
                    const selectedDate = new Date(this.formattedDate);
                    if (!isNaN(selectedDate.getTime())) {
                        this.currentDate = selectedDate;
                    }
                },

                setDate(dateString) {
                    // Parse the date string and update the date
                    const parts = dateString.split('/');
                    const date = new Date(parts[2], parts[1] - 1, parts[0]); // Convert to Date object
                    this.currentDate = date;
                    this.updateFormattedDate();
                },

                openCalendar() {
                    // Make the input visible temporarily
                    this.$refs.dateInput.classList.remove('hidden');
                    // Trigger input click to open the calendar
                    this.$refs.dateInput.click();
                    // Hide the input again after a short delay
                    setTimeout(() => {
                        this.$refs.dateInput.classList.add('hidden');
                    }, 100);
                },

                prevDay() {
                    this.currentDate.setDate(this.currentDate.getDate() - 1);
                    const options = {
                        day: '2-digit',
                        month: '2-digit',
                        year: 'numeric'
                    };
                    this.formattedDate = this.currentDate.toLocaleDateString('en-GB', options);
                    this.updateFormattedDate();
                    getDataByDate(this.formattedDate);
                },

                nextDay() {
                    const tomorrow = new Date(this.currentDate);
                    tomorrow.setDate(this.currentDate.getDate() + 1);
                    this.currentDate = tomorrow;
                    const options = {
                        day: '2-digit',
                        month: '2-digit',
                        year: 'numeric'
                    };
                    this.formattedDate = this.currentDate.toLocaleDateString('en-GB', options);
                    this.updateFormattedDate();
                    getDataByDate(this.formattedDate);
                },
                get isFutureDate() {
                    const tomorrow = new Date(this.currentDate);
                    tomorrow.setDate(this.currentDate.getDate() + 1);
                    return tomorrow > new Date();
                },
            };
        }
